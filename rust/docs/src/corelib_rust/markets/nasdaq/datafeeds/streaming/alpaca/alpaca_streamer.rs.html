108/// Internal state holder for the Alpaca streamer logic.
109struct Inner<C: AlpacaCallbacks> {
110    /// The persistent local database instance.
111    db: Database,
112    /// List of current symbol subscriptions.
113    subscriptions: Vec<String>,
114    /// Configured silence threshold for reconnections.
115    silence_seconds: u32,
116    /// Configuration mapping for the active instance.
117    config: AlpacaConfig,
118    /// The callback implementation.
119    callbacks: C,

121    stop_tx: Option<mpsc::Sender<()>>,
122    /// Channel for sending new symbol subscriptions to the active stream.

124    /// Join handle for the active WebSocket task.

128/// Represents the possible outcomes of the internal websocket loop.
129enum WsLoopResult {
130    /// The loop exited gracefully (e.g., manual stop).
131    GracefulStop,
132    /// The loop encountered a network or protocol error and should retry.
133    Reconnect,
134    /// The loop encountered a fatal error (like Auth Failure) and should halt entirely.
135    FatalError(String),
136}

144impl<C: AlpacaCallbacks> AlpacaStreamingCore<C> {
145    /// Creates a new `AlpacaStreamingCore` and initializes the local database.
146    pub fn new(callbacks: C) -> Self {
147        // Determine the database path from environment or use a temporary file
148        let db_path = std::env::var("ALPACA_DB").unwrap_or_else(|_| {
149            let temp = std::env::temp_dir();
150            temp.join("corelib_streaming.redb")
151                .to_string_lossy()
152                .to_string()
153        });
154
155        // Open or create the redb database
156        let db = Database::create(&db_path).expect("Failed to open redb");
157
158        // Ensure the alpaca_subscriptions table exists in the database
159        {
160            let write_txn = db.begin_write().unwrap();
161            {
162                let _ = write_txn.open_table(ALPACA_SUBSCRIPTIONS_TABLE).unwrap();
163            }
164            write_txn.commit().unwrap();
165        }
166
167        // Load existing subscriptions from the database into memory
168        let read_txn = db.begin_read().unwrap();
169        let table = read_txn.open_table(ALPACA_SUBSCRIPTIONS_TABLE).unwrap();
170        let subscriptions = table
171            .iter()
172            .unwrap()
173            .map(|item| {
174                let (k, _) = item.unwrap();
175                k.value().to_string()
176            })
177            .collect();
178
179        Self {
180            inner: Arc::new(Mutex::new(Inner {
181                db,
182                subscriptions,
183                silence_seconds: 60, // Default to 60s silence check
184                config: AlpacaConfig {
185                    db_path: None,
186                    silence_seconds: None,
187                    base_url: None,
188                    key_id: None,
189                    secret_key: None,
190                },
191                callbacks,
192                stop_tx: None,
193                sub_tx: None,
194                ws_task: None,
195            })),
196        }
197    }

212    /// Spawns the supervisor loop and starts the price stream.
213    pub async fn start(&self) {
214        let inner = Arc::clone(&self.inner);
215        let mut guard = inner.lock().await;
216
217        // Prevent multiple concurrent tasks from running
218        if guard.ws_task.is_some() {
219            return;
220        }
221
222        // Initialize communication channels
223        let (stop_tx, stop_rx) = mpsc::channel(1);
224        let (sub_tx, sub_rx) = mpsc::channel(10);
225
226        guard.stop_tx = Some(stop_tx);
227        guard.sub_tx = Some(sub_tx);
228
229        // Spawn the main supervisor task
230        let task = tokio::spawn(Self::run_loop(Arc::clone(&inner), stop_rx, sub_rx));
231        guard.ws_task = Some(task);
232    }

234    /// Background loop that handles reconnections with exponential backoff.
235    async fn run_loop(
236        inner: Arc<Mutex<Inner<C>>>,
237        mut stop_rx: mpsc::Receiver<()>,
238        mut sub_rx: mpsc::Receiver<Vec<String>>,
239    ) {
240        // Initial backoff duration in seconds
241        let mut backoff = 5u64;
242
243        loop {
244            let inner_clone = Arc::clone(&inner);
245
246            // Execute the actual WebSocket logic
247            let res = Self::ws_loop(inner_clone, &mut sub_rx, &mut stop_rx).await;
248
249            match res {
250                WsLoopResult::GracefulStop => {
251                    // Stream stopped gracefully via stop() call
252                    break;
253                }
254                WsLoopResult::FatalError(err_msg) => {
255                    // A fatal error (like Auth Failure) occurred. Halt the supervisor.
256                    let guard = inner.lock().await;
257                    guard.callbacks.on_log(LogRecord {
258                        level: "fatal".to_string(),
259                        msg: "Streamer halted due to fatal error".to_string(),
260                        extras: Some(err_msg),
261                    });
262                    break;
263                }
264                WsLoopResult::Reconnect => {
265                    // Connection lost or non-fatal error occurred, trigger a reconnect event
266                    inner.lock().await.callbacks.on_event(EventRecord {
267                        r#type: "reconnecting".to_string(),
268                        data: None,
269                    });
270                }
271            }
272
273            // Sleep for the backoff duration before the next reconnect attempt
274            tokio::time::sleep(tokio::time::Duration::from_secs(backoff)).await;
275
276            // Double the backoff duration up to a maximum of 1 hour
277            backoff = (backoff * 2).min(3600);
278        }
279    }

281    /// Handles the active WebSocket connection, authentication, and message dispatching.
282    async fn ws_loop(
283        inner: Arc<Mutex<Inner<C>>>,
284        sub_rx: &mut mpsc::Receiver<Vec<String>>,
285        stop_rx: &mut mpsc::Receiver<()>,
286    ) -> WsLoopResult {
287        // 1. Resolve URL and Credentials
288        let (url, key_id, secret_key) = {
289            let guard = inner.lock().await;
290
291            let url = guard
292                .config
293                .base_url
294                .clone()
295                .or_else(|| std::env::var("APCA_API_BASE_URL").ok())
296                .unwrap_or_else(|| DEFAULT_ALPACA_WS_URL.to_string());
297
298            let key_id = guard
299                .config
300                .key_id
301                .clone()
302                .or_else(|| std::env::var("APCA_API_KEY_ID").ok())
303                .unwrap_or_default();
304
305            let secret_key = guard
306                .config
307                .secret_key
308                .clone()
309                .or_else(|| std::env::var("APCA_API_SECRET_KEY").ok())
310                .unwrap_or_default();
311
312            (url, key_id, secret_key)
313        };
314
315        // 2. Attempt Connection
316        let (mut ws_stream, _) = match connect_async(&url).await {
317            Ok(v) => v,
318            Err(e) => {
319                inner.lock().await.callbacks.on_log(LogRecord {
320                    level: "error".to_string(),
321                    msg: "Alpaca WS connect failed".to_string(),
322                    extras: Some(e.to_string()),
323                });
324                return WsLoopResult::Reconnect;
325            }
326        };
327
328        // 3. Handle Initial Connection Message
329        if let Some(Ok(Message::Text(msg))) = ws_stream.next().await {
330            if let Ok(parsed) = serde_json::from_str::<serde_json::Value>(&msg) {
331                if let Some(arr) = parsed.as_array() {
332                    if let Some(first) = arr.first() {
333                        if first.get("T").and_then(|t| t.as_str()) != Some("success")
334                            || first.get("msg").and_then(|m| m.as_str()) != Some("connected")
335                        {
336                            return WsLoopResult::Reconnect;
337                        }
338                    }
339                }
340            }
341        }
342
343        // 4. Send Authentication Payload
344        let auth_payload = serde_json::json!({
345            "action": "auth",
346            "key": key_id,
347            "secret": secret_key
348        });
349
350        if let Err(e) = ws_stream
351            .send(Message::Text(auth_payload.to_string().into()))
352            .await
353        {
354            inner.lock().await.callbacks.on_log(LogRecord {
355                level: "error".to_string(),
356                msg: "Failed to send auth payload".to_string(),
357                extras: Some(e.to_string()),
358            });
359            return WsLoopResult::Reconnect;
360        }
361
362        // 5. Await Authentication Response
363        if let Some(Ok(Message::Text(msg))) = ws_stream.next().await {
364            if let Ok(parsed) = serde_json::from_str::<serde_json::Value>(&msg) {
365                if let Some(arr) = parsed.as_array() {
366                    if let Some(first) = arr.first() {
367                        // Check for Auth Failure (Fatal)
368                        if first.get("T").and_then(|t| t.as_str()) == Some("error") {
369                            let err_msg = first
370                                .get("msg")
371                                .and_then(|m| m.as_str())
372                                .unwrap_or("Unknown Auth Error")
373                                .to_string();
374                            let code = first.get("code").and_then(|c| c.as_i64()).unwrap_or(0);
375
376                            inner.lock().await.callbacks.on_event(EventRecord {
377                                r#type: "error".to_string(),
378                                data: Some(format!("Auth Failed: {} (Code {})", err_msg, code)),
379                            });
380
381                            return WsLoopResult::FatalError(format!(
382                                "Auth Failed: {} (Code {})",
383                                err_msg, code
384                            ));
385                        }
386
387                        // Check for Auth Success
388                        if first.get("T").and_then(|t| t.as_str()) != Some("success")
389                            || first.get("msg").and_then(|m| m.as_str()) != Some("authenticated")
390                        {
391                            return WsLoopResult::Reconnect;
392                        }
393                    }
394                }
395            }
396        } else {
397            return WsLoopResult::Reconnect;
398        }
399
400        // Notify that the connection has been successfully established and authenticated
401        inner.lock().await.callbacks.on_event(EventRecord {
402            r#type: "connected".to_string(),
403            data: None,
404        });
405
406        // Split the stream into a writer and a reader for concurrent operation
407        let (mut write, mut read) = ws_stream.split();
408
409        // Send the initial subscription payload for all currently tracked symbols
410        {
411            let guard = inner.lock().await;
412            if !guard.subscriptions.is_empty() {
413                // By default, we subscribe to 'quotes' for provided symbols.
414                // The struct can be extended to support trades/bars dynamically.
415                let payload = serde_json::json!({
416                    "action": "subscribe",
417                    "quotes": guard.subscriptions
418                })
419                .to_string();
420
421                let _ = write.send(Message::Text(payload.into())).await;
422            }
423        }
424
425        // Initialize timers for silence detection and WebSocket pings
426        let mut silence_timer = tokio::time::interval(tokio::time::Duration::from_secs(60));
427        let mut ping_timer = tokio::time::interval(tokio::time::Duration::from_secs(PING_INTERVAL));
428
429        // Consume the immediate first ticks of the intervals
430        let _ = silence_timer.tick().await;
431        let _ = ping_timer.tick().await;
432
433        loop {
434            // Select over various input sources concurrently
435            tokio::select! {
436                // Handle graceful stop signal
437                _ = stop_rx.recv() => {
438                    inner.lock().await.callbacks.on_event(EventRecord { r#type: "disconnected".to_string(), data: None });
439                    return WsLoopResult::GracefulStop;
440                }
441
442                // Handle incoming WebSocket messages
443                msg = read.next() => {
444                    match msg {
445                        Some(Ok(Message::Text(text))) => {
446                            // Reset silence timer as we received data
447                            silence_timer.reset();
448
449                            // Parse the message array from Alpaca
450                            let items: Vec<serde_json::Value> = match serde_json::from_str(&text) {
451                                Ok(v) => v,
452                                Err(_) => continue,
453                            };
454
455                            for obj in items {
456                                let t_type = obj.get("T").and_then(|t| t.as_str()).unwrap_or("");
457
458                                match t_type {
459                                    // Process Quotes
460                                    "q" => {
461                                        let data = AlpacaPricingData {
462                                            symbol: obj.get("S").and_then(|s| s.as_str()).unwrap_or("").to_string(),
463                                            message_type: "quote".to_string(),
464                                            price: obj.get("bp").and_then(|v| v.as_f64()).unwrap_or(0.0), // Fallback to bid as primary price
465                                            bid_price: obj.get("bp").and_then(|v| v.as_f64()).unwrap_or(0.0),
466                                            ask_price: obj.get("ap").and_then(|v| v.as_f64()).unwrap_or(0.0),
467                                            volume: obj.get("bs").and_then(|v| v.as_f64()).unwrap_or(0.0), // Bid size as volume proxy
468                                            timestamp: obj.get("t").and_then(|t| t.as_str()).unwrap_or("").to_string(),
469                                        };
470                                        inner.lock().await.callbacks.on_pricing(data);
471                                    }
472                                    // Process Trades
473                                    "t" => {
474                                        let data = AlpacaPricingData {
475                                            symbol: obj.get("S").and_then(|s| s.as_str()).unwrap_or("").to_string(),
476                                            message_type: "trade".to_string(),
477                                            price: obj.get("p").and_then(|v| v.as_f64()).unwrap_or(0.0),
478                                            bid_price: 0.0,
479                                            ask_price: 0.0,
480                                            volume: obj.get("s").and_then(|v| v.as_f64()).unwrap_or(0.0),
481                                            timestamp: obj.get("t").and_then(|t| t.as_str()).unwrap_or("").to_string(),
482                                        };
483                                        inner.lock().await.callbacks.on_pricing(data);
484                                    }
485                                    // Process Bars
486                                    "b" => {
487                                        let data = AlpacaPricingData {
488                                            symbol: obj.get("S").and_then(|s| s.as_str()).unwrap_or("").to_string(),
489                                            message_type: "bar".to_string(),
490                                            price: obj.get("c").and_then(|v| v.as_f64()).unwrap_or(0.0), // Close price
491                                            bid_price: 0.0,
492                                            ask_price: 0.0,
493                                            volume: obj.get("v").and_then(|v| v.as_f64()).unwrap_or(0.0),
494                                            timestamp: obj.get("t").and_then(|t| t.as_str()).unwrap_or("").to_string(),
495                                        };
496                                        inner.lock().await.callbacks.on_pricing(data);
497                                    }
498                                    // Handle subscription success messages
499                                    "subscription" => {
500                                        inner.lock().await.callbacks.on_log(LogRecord {
501                                            level: "info".to_string(),
502                                            msg: "Alpaca subscription updated".to_string(),
503                                            extras: Some(text.to_string()),
504                                        });
505                                    }
506                                    // Handle server errors
507                                    "error" => {
508                                        inner.lock().await.callbacks.on_log(LogRecord {
509                                            level: "error".to_string(),
510                                            msg: "Alpaca API Error".to_string(),
511                                            extras: Some(text.to_string()),
512                                        });
513                                    }
514                                    _ => {
515                                        // Unhandled message types are traced
516                                        inner.lock().await.callbacks.on_log(LogRecord {
517                                            level: "trace".to_string(),
518                                            msg: "Unhandled Alpaca Message".to_string(),
519                                            extras: Some(text.to_string()),
520                                        });
521                                    }
522                                }
523                            }
524                        }
525
526                        // Handle WebSocket close frames
527                        Some(Ok(Message::Close(c))) => {
528                            let data = c.map(|frame| frame.reason.to_string());
529                            inner.lock().await.callbacks.on_event(EventRecord { r#type: "disconnected".to_string(), data });
530                            return WsLoopResult::Reconnect;
531                        }
532
533                        // Handle unexpected end of stream
534                        None => {
535                            inner.lock().await.callbacks.on_event(EventRecord { r#type: "disconnected".to_string(), data: Some("Stream ended".to_string()) });
536                            return WsLoopResult::Reconnect;
537                        }
538
539                        // Handle WebSocket errors
540                        Some(Err(e)) => {
541                            let err_msg = e.to_string();
542                            inner.lock().await.callbacks.on_log(LogRecord { level: "error".to_string(), msg: "WS read error".to_string(), extras: Some(err_msg.clone()) });
543                            inner.lock().await.callbacks.on_event(EventRecord { r#type: "error".to_string(), data: Some(err_msg) });
544                            return WsLoopResult::Reconnect;
545                        }
546                        _ => continue,
547                    }
548                }
549
550                // Handle new symbol subscriptions added while the stream is active
551                new_subs = sub_rx.recv() => {
552                    if let Some(subs) = new_subs {
553                        if !subs.is_empty() {
554                            // Construct and send the subscription message
555                            let payload = serde_json::json!({
556                                "action": "subscribe",
557                                "quotes": subs
558                            }).to_string();
559                            let _ = write.send(Message::Text(payload.into())).await;
560                        }
561                    }
562                }
563
564                // Send periodic ping frames to the server
565                _ = ping_timer.tick() => {
566                    let _ = write.send(Message::Ping(vec![].into())).await;
567                }
568
569                // Reconnect if no data has been received for the silence threshold
570                _ = silence_timer.tick() => {
571                    inner.lock().await.callbacks.on_event(EventRecord { r#type: "silence-reconnect".to_string(), data: None });
572                    return WsLoopResult::Reconnect;
573                }
574            }
575        }
576    }

578    /// Subscribes to a list of symbols and persists them in the database.
579    pub async fn subscribe(&self, symbols: Vec<String>) {
580        let mut guard = self.inner.lock().await;
581        let mut to_send = Vec::new();
582
583        for s in &symbols {
584            // Only add and persist if not already subscribed
585            if !guard.subscriptions.contains(s) {
586                guard.subscriptions.push(s.clone());
587                to_send.push(s.clone());
588
589                // Persist the new subscription in redb
590                let write_txn = guard.db.begin_write().unwrap();
591                {
592                    let mut table = write_txn.open_table(ALPACA_SUBSCRIPTIONS_TABLE).unwrap();
593                    table.insert(s.as_str(), true).unwrap();
594                }
595                write_txn.commit().unwrap();
596            }
597        }
598
599        // If the background task is running, send the new symbols via the channel
600        if let Some(tx) = &guard.sub_tx {
601            let _ = tx.send(to_send).await;
602        }
603    }

605    /// Unsubscribes from a list of symbols and removes them from the database.
606    pub async fn unsubscribe(&self, symbols: Vec<String>) {
607        let mut guard = self.inner.lock().await;
608
609        // Remove symbols from memory
610        guard.subscriptions.retain(|s| !symbols.contains(s));
611
612        // Remove symbols from the persistent database
613        let write_txn = guard.db.begin_write().unwrap();
614        {
615            let mut table = write_txn.open_table(ALPACA_SUBSCRIPTIONS_TABLE).unwrap();
616            for s in &symbols {
617                table.remove(s.as_str()).unwrap();
618            }
619        }
620        write_txn.commit().unwrap();
621
622        // Send the explicit unsubscribe message if the loop is active
623        if let Some(_tx) = &guard.ws_task {
624            // We bypass standard channels here and use the fact that the next reconnect
625            // will drop them, or we could add an unsub channel in the future.
626            // For now, we log the removal.
627            guard.callbacks.on_log(LogRecord {
628                level: "info".to_string(),
629                msg: format!("Unsubscribed from {} symbols", symbols.len()),
630                extras: None,
631            });
632        }
633    }
634
635    /// Clears all subscriptions and stops the streamer.
636    pub async fn clean(&self) {
637        let mut guard = self.inner.lock().await;
638
639        // Delete the entire subscriptions table
640        let write_txn = guard.db.begin_write().unwrap();
641        let _ = write_txn.delete_table(ALPACA_SUBSCRIPTIONS_TABLE);
642        write_txn.commit().unwrap();
643
644        // Reset in-memory state
645        guard.subscriptions.clear();
646
647        // Stop the active task
648        if let Some(tx) = guard.stop_tx.take() {
649            let _ = tx.send(()).await;
650        }
651    }
652
653    /// Stops the background task without clearing persistent subscriptions.
654    pub async fn stop(&self) {
655        let mut guard = self.inner.lock().await;
656
657        // Signal the task to stop
658        if let Some(tx) = guard.stop_tx.take() {
659            let _ = tx.send(()).await;
660        }
661
662        // Force abort the task if it doesn't respond to the signal
663        if let Some(task) = guard.ws_task.take() {
664            task.abort();
665        }
666    }
667}

706    }

727    }
