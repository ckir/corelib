19/// Standardized structure for serialized HTTP responses.
20///
21/// Mirrors the `SerializedResponse<T>` interface from TypeScript.
22#[derive(Debug, Clone, Serialize, Deserialize, PartialEq)]
23pub struct SerializedResponse<T> {
24    /// True if the HTTP status code is 2xx.
25    pub ok: bool,
26    /// The numeric HTTP status code.
27    pub status: u16,
28    /// The status text or reason phrase returned by the server.
29    pub status_text: String,
30    /// A map of response headers.
31    pub headers: HashMap<String, String>,
32    /// The final URL reached after redirects.
33    pub url: String,
34    /// The successfully parsed response body of type `T`.
35    pub body: T,
36}
37
38/// Discriminated union for API results, providing a consistent success/error structure.
39///
40/// It maps exactly to `{"status": "success", "value": ...}` or `{"status": "error", "reason": ...}`
41/// for seamless serialization between Rust and TypeScript.
42#[derive(Debug, Clone, Serialize, Deserialize, PartialEq)]
43#[serde(tag = "status", rename_all = "lowercase")]
44pub enum ApiResponse<T> {
45    /// Represents a successful HTTP request with a 2xx status code and successfully parsed body.
46    Success {
47        /// The serialized response wrapper containing the parsed body and metadata.
48        value: SerializedResponse<T>,
49    },
50    /// Represents a failure, either at the network level, HTTP level (non-2xx), or parsing level.
51    Error {
52        /// Detailed reason for the error, structured as a JSON value for cross-language compatibility.
53        reason: Value,
54    },
55}

57/// Request configuration options for high-resilience fetching.
58///
59/// Defines overrides for standard HTTP parameters and the underlying `ky` retry logic.
60#[derive(Debug, Clone, Default)]
61pub struct RequestOptions {
62    /// The HTTP method to use (Defaults to GET).
63    pub method: Option<Method>,
64    /// A map of custom HTTP headers to include in the request.
65    pub headers: Option<HashMap<String, String>>,
66    /// A JSON value to be sent as the request body (for POST/PUT/PATCH).
67    pub json: Option<Value>,
68    /// An optional timeout duration for the entire request.
69    pub timeout: Option<Duration>,
70    /// The maximum number of retry attempts allowed for this request.
71    pub retry_limit: Option<u32>,
72}

74/// Makes an HTTP request to a single URL with built-in resilience features.
75///
76/// This function uses the underlying `ky` implementation to automatically handle
77/// retries and timeouts based on the provided options or defaults.
78///
79/// # Arguments
80/// * `url` - The target URL to fetch.
81/// * `options` - Optional configuration overrides (method, headers, body, retries).
82///
83/// # Returns
84/// An `ApiResponse<T>` containing either the `SerializedResponse` on success
85/// or a JSON-structured error reason on failure.
86pub async fn end_point<T: DeserializeOwned>(
87    url: &str,
88    options: Option<RequestOptions>,
89) -> ApiResponse<T> {
90    // Extract options or use defaults
91    let opts = options.unwrap_or_default();
92    let method = opts.method.unwrap_or(Method::GET);
93
94    // Initialize the appropriate ky builder based on the specified HTTP method
95    let mut builder = match method {
96        Method::POST => ky::post(url),
97        Method::PUT => ky::put(url),
98        Method::PATCH => ky::patch(url),
99        Method::DELETE => ky::delete(url),
100        _ => ky::get(url),
101    };
102
103    // Apply custom headers if provided
104    if let Some(headers) = opts.headers {
105        for (k, v) in headers {
106            builder = builder.header(k, v);
107        }
108    }
109
110    // Attach the JSON payload if present
111    if let Some(json_body) = opts.json {
112        builder = builder.json(&json_body);
113    }
114
115    // Override the default timeout if specified
116    if let Some(t) = opts.timeout {
117        builder = builder.timeout(t);
118    }
119
120    // Override the default retry limit if specified
121    if let Some(r) = opts.retry_limit {
122        builder = builder.retry(r);
123    }
124
125    // Execute the request via the ky client
126    let result = builder.send().await;
127
128    match result {
129        Ok(ky_res) => {
130            // Successfully received a response (2xx). Extract properties before consuming the body.
131            let inner_res = ky_res.into_inner();
132            let status = inner_res.status();
133            let ok = status.is_success();
134            let status_code = status.as_u16();
135            let status_text = status.canonical_reason().unwrap_or("").to_string();
136            let res_url = inner_res.url().to_string();
137
138            // Map headers into a standard HashMap
139            let mut headers = HashMap::new();
140            for (k, v) in inner_res.headers() {
141                headers.insert(k.as_str().to_string(), v.to_str().unwrap_or("").to_string());
142            }
143
144            // Attempt to deserialize the response body as the requested type T
145            match inner_res.json::<T>().await {
146                Ok(body) => ApiResponse::Success {
147                    value: SerializedResponse {
148                        ok,
149                        status: status_code,
150                        status_text,
151                        headers,
152                        url: res_url,
153                        body,
154                    },
155                },
156                Err(e) => ApiResponse::Error {
157                    // Body was not valid JSON or didn't match the expected schema
158                    reason: serde_json::json!({
159                        "message": "Failed to parse response body",
160                        "error": e.to_string(),
161                        "status": status_code,
162                    }),
163                },
164            }
165        }
166        Err(e) => {
167            // The request failed at the HTTP level (non-2xx) or transport level.
168            // Map the ky::KyError into a consistent JSON reason.
169            let reason = match e {
170                ky::KyError::Http { status, url } => {
171                    serde_json::json!({
172                        "message": format!("HTTP Error {}", status),
173                        "status": status.as_u16(),
174                        "url": url,
175                    })
176                }
177                ky::KyError::Request(req_err) => {
178                    serde_json::json!({
179                        "message": "Transport Error",
180                        "error": req_err.to_string(),
181                    })
182                }
183                ky::KyError::Serialization(ser_err) => {
184                    serde_json::json!({
185                        "message": "Serialization Error",
186                        "error": ser_err.to_string(),
187                    })
188                }
189            };
190
191            ApiResponse::Error { reason }
192        }
193    }
194}

196/// Makes parallel HTTP requests to multiple URLs concurrently.
197///
198/// All requests are executed simultaneously using `futures::future::join_all`.
199/// The order of results in the returned vector matches the order of input URLs.
200///
201/// # Arguments
202/// * `urls` - A slice of target URLs to fetch.
203/// * `options` - Shared configuration overrides applied to every request in the batch.
204///
205/// # Returns
206/// A vector of `ApiResponse<T>` objects.
207pub async fn end_points<T: DeserializeOwned>(
208    urls: &[&str],
209    options: Option<RequestOptions>,
210) -> Vec<ApiResponse<T>> {
211    let opts = options.unwrap_or_default();
212
213    // Create an iterator of futures, each calling end_point for a specific URL
214    let futures = urls.iter().map(|&url| {
215        // Clone the shared options for each individual task
216        let cloned_opts = opts.clone();
217        async move { end_point::<T>(url, Some(cloned_opts)).await }
218    });
219
220    // Execute all futures concurrently and await their completion
221    join_all(futures).await
222}
