9/// A Rust HTTP client that mimics the `ky` API for ergonomics.
10pub mod ky;
11
12/// High-resilience HTTP request utility for single or multiple endpoints.
13pub mod unlimited;
14
15/// High-resilience proxied HTTP client with automatic rotation and fallback.
16pub mod proxied;
17
18// Re-export the public API so users can write:
19// use corelib_rust::retrieve::ky;
20
21/// Shorthand function for a DELETE request.
22pub use ky::delete;
23/// Shorthand function for a GET request.
24pub use ky::get;
25/// Shorthand function for a PATCH request.
26pub use ky::patch;
27/// Shorthand function for a POST request.
28pub use ky::post;
29/// Shorthand function for a PUT request.
30pub use ky::put;
31/// The main `Ky` instance used for making requests.
32pub use ky::Ky;
33/// Error type for `ky` requests.
34pub use ky::KyError;
35/// Chainable builder for `ky` requests.
36pub use ky::KyRequestBuilder;
37/// Wrapper around `reqwest::Response` with extra helpers.
38pub use ky::KyResponse;
39
40// Re-export the public API so users can write:
41// use corelib_rust::retrieve::unlimited;
42
43/// High-level function to make a single resilient request.
44pub use unlimited::end_point;

46pub use unlimited::end_points;
