23/// The default production WebSocket URL for the Alpaca IEX data stream.
24const DEFAULT_ALPACA_WS_URL: &str = "wss://stream.data.alpaca.markets/v2/iex";

26/// Interval in seconds for sending WebSocket ping frames to keep the connection alive.

70/// A generic trait for handling Alpaca streamer callbacks.
71pub trait AlpacaCallbacks: Send + Sync + 'static {
72    /// Called when the streamer emits a log message.
73    fn on_log(&self, record: LogRecord);
74    /// Called when a new price update is successfully decoded.
75    fn on_pricing(&self, data: AlpacaPricingData);
76    /// Called when a lifecycle event occurs (connection state changes).
77    fn on_event(&self, event: EventRecord);
78}

108/// Internal state holder for the Alpaca streamer logic.
109struct Inner<C: AlpacaCallbacks> {
110    /// The persistent local database instance.
111    db: Option<Database>,
112    /// List of current symbol subscriptions.
113    subscriptions: Vec<String>,
114    /// Configured silence threshold for reconnections.
115    silence_seconds: u32,
116    /// Configuration mapping for the active instance.
117    config: AlpacaConfig,
118    /// The callback implementation.
119    callbacks: C,
120    /// Channel for sending a stop signal to the background task.
121    stop_tx: Option<mpsc::Sender<()>>,
122    /// Channel for sending new symbol subscriptions to the active stream.
123    sub_tx: Option<mpsc::Sender<Vec<String>>>,
124    /// Join handle for the active WebSocket task.
125    ws_task: Option<tokio::task::JoinHandle<()>>,
126}

128/// Represents the possible outcomes of the internal websocket loop.
129enum WsLoopResult {
130    /// The loop exited gracefully (e.g., manual stop).
131    GracefulStop,
132    /// The loop encountered a network or protocol error and should retry.
133    Reconnect,
134    /// The loop encountered a fatal error (like Auth Failure) and should halt entirely.
135    FatalError(String),
136}

144impl<C: AlpacaCallbacks> AlpacaStreamingCore<C> {
145    /// Creates a new `AlpacaStreamingCore` and optionally initializes the local database.
146    pub fn new(callbacks: C) -> Self {
147        // Determine the database path from environment or use a temporary file
148        let db_env = std::env::var("ALPACA_DB").unwrap_or_else(|_| {
149            let temp = std::env::temp_dir();
150            temp.join("corelib_streaming.redb")
151                .to_string_lossy()
152                .to_string()
153        });
154
155        let (db, subscriptions) = if db_env == "NOT_SET" {
156            (None, Vec::new())
157        } else {
158            // Open or create the redb database
159            let db = Database::create(&db_env).expect("Failed to open redb");
160
161            // Ensure the alpaca_subscriptions table exists in the database
162            {
163                let write_txn = db.begin_write().unwrap();
164                {
165                    let _ = write_txn.open_table(ALPACA_SUBSCRIPTIONS_TABLE).unwrap();
166                }
167                write_txn.commit().unwrap();
168            }
169
170            // Load existing subscriptions from the database into memory
171            let read_txn = db.begin_read().unwrap();
172            let table = read_txn.open_table(ALPACA_SUBSCRIPTIONS_TABLE).unwrap();
173            let loaded_subscriptions = table
174                .iter()
175                .unwrap()
176                .map(|item| {
177                    let (k, _) = item.unwrap();
178                    k.value().to_string()
179                })
180                .collect();
181            (Some(db), loaded_subscriptions)
182        };
183
184        Self {
185            inner: Arc::new(Mutex::new(Inner {
186                db,
187                subscriptions,
188                silence_seconds: 60, // Default to 60s silence check
189                config: AlpacaConfig {
190                    db_path: None,
191                    silence_seconds: None,
192                    base_url: None,
193                    key_id: None,
194                    secret_key: None,
195                },
196                callbacks,
197                stop_tx: None,
198                sub_tx: None,
199                ws_task: None,
200            })),
201        }
202    }

204    /// Initializes the streamer with the provided configuration.
205    pub async fn init(&self, config: AlpacaConfig) {
206        let mut inner = self.inner.lock().await;
207
208        // Override the silence threshold if provided
209        if let Some(s) = config.silence_seconds {

213        // Store the config for authentication use in the loop
214        inner.config = config;

217    /// Spawns the supervisor loop and starts the price stream.
218    pub async fn start(&self) {
219        let inner = Arc::clone(&self.inner);
220        let mut guard = inner.lock().await;
221
222        // Prevent multiple concurrent tasks from running
223        if guard.ws_task.is_some() {
224            return;
225        }
226
227        // Initialize communication channels
228        let (stop_tx, stop_rx) = mpsc::channel(1);
229        let (sub_tx, sub_rx) = mpsc::channel(10);
230
231        guard.stop_tx = Some(stop_tx);
232        guard.sub_tx = Some(sub_tx);
233
234        // Spawn the main supervisor task
235        let task = tokio::spawn(Self::run_loop(Arc::clone(&inner), stop_rx, sub_rx));
236        guard.ws_task = Some(task);
237    }

239    /// Background loop that handles reconnections with exponential backoff.
240    async fn run_loop(
241        inner: Arc<Mutex<Inner<C>>>,
242        mut stop_rx: mpsc::Receiver<()>,
243        mut sub_rx: mpsc::Receiver<Vec<String>>,
244    ) {
245        // Initial backoff duration in seconds
246        let mut backoff = 5u64;
247
248        loop {
249            let inner_clone = Arc::clone(&inner);
250
251            // Execute the actual WebSocket logic
252            let res = Self::ws_loop(inner_clone, &mut sub_rx, &mut stop_rx).await;
253
254            match res {
255                WsLoopResult::GracefulStop => {
256                    // Stream stopped gracefully via stop() call
257                    break;
258                }
259                WsLoopResult::FatalError(err_msg) => {
260                    // A fatal error (like Auth Failure) occurred. Halt the supervisor.
261                    let guard = inner.lock().await;
262                    guard.callbacks.on_log(LogRecord {
263                        level: "fatal".to_string(),
264                        msg: "Streamer halted due to fatal error".to_string(),
265                        extras: Some(err_msg),
266                    });
267                    break;
268                }
269                WsLoopResult::Reconnect => {
270                    // Connection lost or non-fatal error occurred, trigger a reconnect event
271                    inner.lock().await.callbacks.on_event(EventRecord {
272                        r#type: "reconnecting".to_string(),
273                        data: None,
274                    });
275                }
276            }
277
278            // Sleep for the backoff duration before the next reconnect attempt
279            tokio::time::sleep(tokio::time::Duration::from_secs(backoff)).await;
280
281            // Double the backoff duration up to a maximum of 1 hour
282            backoff = (backoff * 2).min(3600);
283        }
284    }

286    /// Handles the active WebSocket connection, authentication, and message dispatching.
287    async fn ws_loop(
288        inner: Arc<Mutex<Inner<C>>>,
289        sub_rx: &mut mpsc::Receiver<Vec<String>>,
290        stop_rx: &mut mpsc::Receiver<()>,
291    ) -> WsLoopResult {
292        // 1. Resolve URL and Credentials
293        let (url, key_id, secret_key) = {
294            let guard = inner.lock().await;
295
296            let url = guard
297                .config
298                .base_url
299                .clone()
300                .or_else(|| std::env::var("APCA_API_BASE_URL").ok())
301                .unwrap_or_else(|| DEFAULT_ALPACA_WS_URL.to_string());
302
303            let key_id = guard
304                .config
305                .key_id
306                .clone()
307                .or_else(|| std::env::var("APCA_API_KEY_ID").ok())
308                .unwrap_or_default();
309
310            let secret_key = guard
311                .config
312                .secret_key
313                .clone()
314                .or_else(|| std::env::var("APCA_API_SECRET_KEY").ok())
315                .unwrap_or_default();
316
317            (url, key_id, secret_key)
318        };
319
320        // 2. Attempt Connection
321        let (mut ws_stream, _) = match connect_async(&url).await {
322            Ok(v) => v,
323            Err(e) => {
324                inner.lock().await.callbacks.on_log(LogRecord {
325                    level: "error".to_string(),
326                    msg: "Alpaca WS connect failed".to_string(),
327                    extras: Some(e.to_string()),
328                });
329                return WsLoopResult::Reconnect;
330            }
331        };
332
333        // 3. Handle Initial Connection Message
334        if let Some(Ok(Message::Text(msg))) = ws_stream.next().await {
335            if let Ok(parsed) = serde_json::from_str::<serde_json::Value>(&msg) {
336                if let Some(arr) = parsed.as_array() {
337                    if let Some(first) = arr.first() {
338                        if first.get("T").and_then(|t| t.as_str()) != Some("success")
339                            || first.get("msg").and_then(|m| m.as_str()) != Some("connected")
340                        {
341                            return WsLoopResult::Reconnect;
342                        }
343                    }
344                }
345            }
346        }
347
348        // 4. Send Authentication Payload
349        let auth_payload = serde_json::json!({
350            "action": "auth",
351            "key": key_id,
352            "secret": secret_key
353        });
354
355        if let Err(e) = ws_stream
356            .send(Message::Text(auth_payload.to_string().into()))
357            .await
358        {
359            inner.lock().await.callbacks.on_log(LogRecord {
360                level: "error".to_string(),
361                msg: "Failed to send auth payload".to_string(),
362                extras: Some(e.to_string()),
363            });
364            return WsLoopResult::Reconnect;
365        }
366
367        // 5. Await Authentication Response
368        if let Some(Ok(Message::Text(msg))) = ws_stream.next().await {
369            if let Ok(parsed) = serde_json::from_str::<serde_json::Value>(&msg) {
370                if let Some(arr) = parsed.as_array() {
371                    if let Some(first) = arr.first() {
372                        // Check for Auth Failure (Fatal)
373                        if first.get("T").and_then(|t| t.as_str()) == Some("error") {
374                            let err_msg = first
375                                .get("msg")
376                                .and_then(|m| m.as_str())
377                                .unwrap_or("Unknown Auth Error")
378                                .to_string();
379                            let code = first.get("code").and_then(|c| c.as_i64()).unwrap_or(0);
380
381                            inner.lock().await.callbacks.on_event(EventRecord {
382                                r#type: "error".to_string(),
383                                data: Some(format!("Auth Failed: {} (Code {})", err_msg, code)),
384                            });
385
386                            return WsLoopResult::FatalError(format!(
387                                "Auth Failed: {} (Code {})",
388                                err_msg, code
389                            ));
390                        }
391
392                        // Check for Auth Success
393                        if first.get("T").and_then(|t| t.as_str()) != Some("success")
394                            || first.get("msg").and_then(|m| m.as_str()) != Some("authenticated")
395                        {
396                            return WsLoopResult::Reconnect;
397                        }
398                    }
399                }
400            }
401        } else {
402            return WsLoopResult::Reconnect;
403        }
404
405        // Notify that the connection has been successfully established and authenticated
406        inner.lock().await.callbacks.on_event(EventRecord {
407            r#type: "connected".to_string(),
408            data: None,
409        });
410
411        // Split the stream into a writer and a reader for concurrent operation
412        let (mut write, mut read) = ws_stream.split();
413
414        // Send the initial subscription payload for all currently tracked symbols
415        {
416            let guard = inner.lock().await;
417            if !guard.subscriptions.is_empty() {
418                // By default, we subscribe to 'quotes' for provided symbols.
419                // The struct can be extended to support trades/bars dynamically.
420                let payload = serde_json::json!({
421                    "action": "subscribe",
422                    "quotes": guard.subscriptions
423                })
424                .to_string();
425
426                let _ = write.send(Message::Text(payload.into())).await;
427            }
428        }
429
430        // Initialize timers for silence detection and WebSocket pings
431        let mut silence_timer = tokio::time::interval(tokio::time::Duration::from_secs(60));
432        let mut ping_timer = tokio::time::interval(tokio::time::Duration::from_secs(PING_INTERVAL));
433
434        // Consume the immediate first ticks of the intervals
435        let _ = silence_timer.tick().await;
436        let _ = ping_timer.tick().await;
437
438        loop {
439            // Select over various input sources concurrently
440            tokio::select! {
441                // Handle graceful stop signal
442                _ = stop_rx.recv() => {
443                    inner.lock().await.callbacks.on_event(EventRecord { r#type: "disconnected".to_string(), data: None });
444                    return WsLoopResult::GracefulStop;
445                }
446
447                // Handle incoming WebSocket messages
448                msg = read.next() => {
449                    match msg {
450                        Some(Ok(Message::Text(text))) => {
451                            // Reset silence timer as we received data
452                            silence_timer.reset();
453
454                            // Parse the message array from Alpaca
455                            let items: Vec<serde_json::Value> = match serde_json::from_str(&text) {
456                                Ok(v) => v,
457                                Err(_) => continue,
458                            };
459
460                            for obj in items {
461                                let t_type = obj.get("T").and_then(|t| t.as_str()).unwrap_or("");
462
463                                match t_type {
464                                    // Process Quotes
465                                    "q" => {
466                                        let data = AlpacaPricingData {
467                                            symbol: obj.get("S").and_then(|s| s.as_str()).unwrap_or("").to_string(),
468                                            message_type: "quote".to_string(),
469                                            price: obj.get("bp").and_then(|v| v.as_f64()).unwrap_or(0.0), // Fallback to bid as primary price
470                                            bid_price: obj.get("bp").and_then(|v| v.as_f64()).unwrap_or(0.0),
471                                            ask_price: obj.get("ap").and_then(|v| v.as_f64()).unwrap_or(0.0),
472                                            volume: obj.get("bs").and_then(|v| v.as_f64()).unwrap_or(0.0), // Bid size as volume proxy
473                                            timestamp: obj.get("t").and_then(|t| t.as_str()).unwrap_or("").to_string(),
474                                        };
475                                        inner.lock().await.callbacks.on_pricing(data);
476                                    }
477                                    // Process Trades
478                                    "t" => {
479                                        let data = AlpacaPricingData {
480                                            symbol: obj.get("S").and_then(|s| s.as_str()).unwrap_or("").to_string(),
481                                            message_type: "trade".to_string(),
482                                            price: obj.get("p").and_then(|v| v.as_f64()).unwrap_or(0.0),
483                                            bid_price: 0.0,
484                                            ask_price: 0.0,
485                                            volume: obj.get("s").and_then(|v| v.as_f64()).unwrap_or(0.0),
486                                            timestamp: obj.get("t").and_then(|t| t.as_str()).unwrap_or("").to_string(),
487                                        };
488                                        inner.lock().await.callbacks.on_pricing(data);
489                                    }
490                                    // Process Bars
491                                    "b" => {
492                                        let data = AlpacaPricingData {
493                                            symbol: obj.get("S").and_then(|s| s.as_str()).unwrap_or("").to_string(),
494                                            message_type: "bar".to_string(),
495                                            price: obj.get("c").and_then(|v| v.as_f64()).unwrap_or(0.0), // Close price
496                                            bid_price: 0.0,
497                                            ask_price: 0.0,
498                                            volume: obj.get("v").and_then(|v| v.as_f64()).unwrap_or(0.0),
499                                            timestamp: obj.get("t").and_then(|t| t.as_str()).unwrap_or("").to_string(),
500                                        };
501                                        inner.lock().await.callbacks.on_pricing(data);
502                                    }
503                                    // Handle subscription success messages
504                                    "subscription" => {
505                                        inner.lock().await.callbacks.on_log(LogRecord {
506                                            level: "info".to_string(),
507                                            msg: "Alpaca subscription updated".to_string(),
508                                            extras: Some(text.to_string()),
509                                        });
510                                    }
511                                    // Handle server errors
512                                    "error" => {
513                                        inner.lock().await.callbacks.on_log(LogRecord {
514                                            level: "error".to_string(),
515                                            msg: "Alpaca API Error".to_string(),
516                                            extras: Some(text.to_string()),
517                                        });
518                                    }
519                                    _ => {
520                                        // Unhandled message types are traced
521                                        inner.lock().await.callbacks.on_log(LogRecord {
522                                            level: "trace".to_string(),
523                                            msg: "Unhandled Alpaca Message".to_string(),
524                                            extras: Some(text.to_string()),
525                                        });
526                                    }
527                                }
528                            }
529                        }
530
531                        // Handle WebSocket close frames
532                        Some(Ok(Message::Close(c))) => {
533                            let data = c.map(|frame| frame.reason.to_string());
534                            inner.lock().await.callbacks.on_event(EventRecord { r#type: "disconnected".to_string(), data });
535                            return WsLoopResult::Reconnect;
536                        }
537
538                        // Handle unexpected end of stream
539                        None => {
540                            inner.lock().await.callbacks.on_event(EventRecord { r#type: "disconnected".to_string(), data: Some("Stream ended".to_string()) });
541                            return WsLoopResult::Reconnect;
542                        }
543
544                        // Handle WebSocket errors
545                        Some(Err(e)) => {
546                            let err_msg = e.to_string();
547                            inner.lock().await.callbacks.on_log(LogRecord { level: "error".to_string(), msg: "WS read error".to_string(), extras: Some(err_msg.clone()) });
548                            inner.lock().await.callbacks.on_event(EventRecord { r#type: "error".to_string(), data: Some(err_msg) });
549                            return WsLoopResult::Reconnect;
550                        }
551                        _ => continue,
552                    }
553                }
554
555                // Handle new symbol subscriptions added while the stream is active
556                new_subs = sub_rx.recv() => {
557                    if let Some(subs) = new_subs {
558                        if !subs.is_empty() {
559                            // Construct and send the subscription message
560                            let payload = serde_json::json!({
561                                "action": "subscribe",
562                                "quotes": subs
563                            }).to_string();
564                            let _ = write.send(Message::Text(payload.into())).await;
565                        }
566                    }
567                }
568
569                // Send periodic ping frames to the server
570                _ = ping_timer.tick() => {
571                    let _ = write.send(Message::Ping(vec![].into())).await;
572                }
573
574                // Reconnect if no data has been received for the silence threshold
575                _ = silence_timer.tick() => {
576                    inner.lock().await.callbacks.on_event(EventRecord { r#type: "silence-reconnect".to_string(), data: None });
577                    return WsLoopResult::Reconnect;
578                }
579            }
580        }
581    }

583    /// Subscribes to a list of symbols and persists them in the database.
584    pub async fn subscribe(&self, symbols: Vec<String>) {
585        let mut guard = self.inner.lock().await;
586        let mut to_send = Vec::new();
587
588        for s in &symbols {
589            // Only add and persist if not already subscribed
590            if !guard.subscriptions.contains(s) {
591                guard.subscriptions.push(s.clone());
592                to_send.push(s.clone());
593
594                // Persist the new subscription in redb if database is active
595                if let Some(ref db) = guard.db {
596                    let write_txn = db.begin_write().unwrap();
597                    {
598                        let mut table = write_txn.open_table(ALPACA_SUBSCRIPTIONS_TABLE).unwrap();
599                        table.insert(s.as_str(), true).unwrap();
600                    }
601                    write_txn.commit().unwrap();
602                }
603            }
604        }
605
606        // If the background task is running, send the new symbols via the channel
607        if let Some(tx) = &guard.sub_tx {
608            let _ = tx.send(to_send).await;
609        }
610    }

612    /// Unsubscribes from a list of symbols and removes them from the database.
613    pub async fn unsubscribe(&self, symbols: Vec<String>) {
614        let mut guard = self.inner.lock().await;
615
616        // Remove symbols from memory
617        guard.subscriptions.retain(|s| !symbols.contains(s));
618
619        // Remove symbols from the persistent database if active
620        if let Some(ref db) = guard.db {
621            let write_txn = db.begin_write().unwrap();
622            {
623                let mut table = write_txn.open_table(ALPACA_SUBSCRIPTIONS_TABLE).unwrap();
624                for s in &symbols {
625                    table.remove(s.as_str()).unwrap();
626                }
627            }
628            write_txn.commit().unwrap();
629        }
630
631        // Send the explicit unsubscribe message if the loop is active
632        if let Some(_tx) = &guard.ws_task {
633            // We bypass standard channels here and use the fact that the next reconnect
634            // will drop them, or we could add an unsub channel in the future.
635            // For now, we log the removal.
636            guard.callbacks.on_log(LogRecord {
637                level: "info".to_string(),
638                msg: format!("Unsubscribed from {} symbols", symbols.len()),
639                extras: None,
640            });
641        }
642    }
643
644    /// Clears all subscriptions and stops the streamer.
645    pub async fn clean(&self) {
646        let mut guard = self.inner.lock().await;
647
648        // Delete the entire subscriptions table if database is active
649        if let Some(ref db) = guard.db {
650            let write_txn = db.begin_write().unwrap();
651            let _ = write_txn.delete_table(ALPACA_SUBSCRIPTIONS_TABLE);
652            write_txn.commit().unwrap();
653        }
654
655        // Reset in-memory state
656        guard.subscriptions.clear();
657
658        // Stop the active task
659        if let Some(tx) = guard.stop_tx.take() {
660            let _ = tx.send(()).await;
661        }
662    }
663
664    /// Stops the background task without clearing persistent subscriptions.
665    pub async fn stop(&self) {
666        let mut guard = self.inner.lock().await;
667
668        // Signal the task to stop
669        if let Some(tx) = guard.stop_tx.take() {
670            let _ = tx.send(()).await;
671        }
672
673        // Force abort the task if it doesn't respond to the signal
674        if let Some(task) = guard.ws_task.take() {
675            task.abort();
676        }
677    }
678}

680/// N-API wrapper for the Alpaca price streamer, enabling its use in JavaScript environments.

683    /// The core implementation using N-API callbacks.
684    core: AlpacaStreamingCore<NapiCallbacks>,
685}
686

688impl AlpacaStreaming {
689    /// Constructs a new `AlpacaStreaming` instance with the provided JavaScript callback functions.

693        on_pricing: ThreadsafeFunction<AlpacaPricingData>,
694        on_event: ThreadsafeFunction<EventRecord>,

702        }

709        Ok(())
710    }

712    /// Starts the long-running streaming task.
713    #[napi]

715        self.core.start().await;
716        Ok(())
717    }

719    /// Adds a list of symbols to the active stream.

722        self.core.subscribe(symbols).await;
723        Ok(())
724    }

730        Ok(())
731    }

737        Ok(())
738    }
