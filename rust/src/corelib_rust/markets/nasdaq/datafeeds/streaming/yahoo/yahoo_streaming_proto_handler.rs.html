14/// The primary data structure for Yahoo Finance real-time quotes.
15///
16/// This struct corresponds to the `PricingData` message in the `.proto` definition.
17/// It contains price, volume, and metadata for a specific financial instrument.
18#[derive(Clone, PartialEq, Message, Serialize, Deserialize)]
19pub struct PricingData {
20    /// The ticker symbol of the instrument (e.g., "AAPL").
21    #[prost(string, tag = "1")]
22    pub id: String,
23    /// The current market price.
24    #[prost(float, tag = "2")]
25    pub price: f32,
26    /// Timestamp of the quote in milliseconds since epoch.
27    #[prost(sint64, tag = "3")]
28    pub time: i64,
29    /// The currency code (e.g., "USD").
30    #[prost(string, tag = "4")]
31    pub currency: String,
32    /// The exchange identifier (e.g., "NMS").
33    #[prost(string, tag = "5")]
34    pub exchange: String,
35    /// The type of quote (Equity, ETF, etc.).
36    #[prost(enumeration = "QuoteType", tag = "6")]
37    pub quote_type: i32,
38    /// The current market session (Regular, Pre, Post).
39    #[prost(enumeration = "MarketHoursType", tag = "7")]
40    pub market_hours: i32,
41    /// Percentage change since the previous close.
42    #[prost(float, tag = "8")]
43    pub change_percent: f32,
44    /// Cumulative volume for the current trading day.
45    #[prost(sint64, tag = "9")]
46    pub day_volume: i64,
47    /// Highest price reached during the current trading day.
48    #[prost(float, tag = "10")]
49    pub day_high: f32,
50    /// Lowest price reached during the current trading day.
51    #[prost(float, tag = "11")]
52    pub day_low: f32,
53    /// Absolute price change since the previous close.
54    #[prost(float, tag = "12")]
55    pub change: f32,
56    /// The short descriptive name of the security.
57    #[prost(string, tag = "13")]
58    pub short_name: String,
59    /// Expiration date for options or futures (milliseconds epoch).
60    #[prost(sint64, tag = "14")]
61    pub expire_date: i64,
62    /// The opening price for the current session.
63    #[prost(float, tag = "15")]
64    pub open_price: f32,
65    /// The closing price of the previous trading session.
66    #[prost(float, tag = "16")]
67    pub previous_close: f32,
68    /// The strike price for options contracts.
69    #[prost(float, tag = "17")]
70    pub strike_price: f32,
71    /// The underlying ticker for derivative contracts.
72    #[prost(string, tag = "18")]
73    pub underlying_symbol: String,
74    /// The total number of outstanding derivative contracts.
75    #[prost(sint64, tag = "19")]
76    pub open_interest: i64,
77    /// The type of option (Call or Put).
78    #[prost(enumeration = "OptionType", tag = "20")]
79    pub option_type: i32,
80}

82/// An N-API compatible object representing pricing data.

98    pub day_high: f64,

100    pub change: f64,
101    pub short_name: String,

103    pub open_price: f64,
104    pub previous_close: f64,
105    pub strike_price: f64,

111impl From<PricingData> for JsPricingData {
112    /// Converts a native `PricingData` struct into a `JsPricingData` object.
113    fn from(p: PricingData) -> Self {
114        Self {
115            id: p.id,
116            price: p.price as f64,
117            time: p.time,
118            currency: p.currency,
119            exchange: p.exchange,
120            quote_type: p.quote_type,
121            market_hours: p.market_hours,
122            change_percent: p.change_percent as f64,
123            day_volume: p.day_volume,
124            day_high: p.day_high as f64,
125            day_low: p.day_low as f64,
126            change: p.change as f64,
127            short_name: p.short_name,
128            expire_date: p.expire_date,
129            open_price: p.open_price as f64,
130            previous_close: p.previous_close as f64,
131            strike_price: p.strike_price as f64,
132            underlying_symbol: p.underlying_symbol,
133            open_interest: p.open_interest,
134            option_type: p.option_type,
135        }
136    }
137}

164/// Represents the specific trading session of the quote.
165#[napi]
166#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash, PartialOrd, Ord, prost::Enumeration)]
167#[repr(i32)]
168pub enum MarketHoursType {
169    /// Before the official market open.
170    PreMarket = 0,
171    /// During official trading hours.
172    RegularMarket = 1,
173    /// After the official market close.
174    PostMarket = 2,
175    /// General catch-all for non-regular hours.
176    ExtendedHoursMarket = 3,
177}

192    use super::*;
193    use prost::Message;

195    #[test]
196    fn test_pricing_data_serialization_cycle() {
197        let original = PricingData {
198            id: "AAPL".to_string(),
199            price: 150.50,
200            time: 1698400000000,
201            currency: "USD".to_string(),
202            exchange: "NMS".to_string(),
203            quote_type: QuoteType::Equity as i32,
204            market_hours: MarketHoursType::RegularMarket as i32,
205            change_percent: 1.5,
206            day_volume: 50000000,
207            day_high: 151.0,
208            day_low: 149.0,
209            change: 2.25,
210            short_name: "Apple Inc.".to_string(),
211            expire_date: 0,
212            open_price: 150.0,
213            previous_close: 148.0,
214            strike_price: 0.0,
215            underlying_symbol: "".to_string(),
216            open_interest: 0,
217            option_type: OptionType::Call as i32,
218        };
219
220        // // Encode to bytes
221        let mut buf = Vec::new();
222        original.encode(&mut buf).unwrap();
223
224        // // Decode back to struct
225        let decoded = PricingData::decode(&buf[..]).unwrap();
226
227        assert_eq!(original.id, decoded.id);
228        assert_eq!(original.price, decoded.price);
229        assert_eq!(original.quote_type, decoded.quote_type);
230    }
231
232    #[test]
233    fn test_enum_conversions() {
234        assert_eq!(QuoteType::Equity as i32, 8);
235        assert_eq!(MarketHoursType::RegularMarket as i32, 1);
236        assert_eq!(OptionType::Call as i32, 0);
237    }
