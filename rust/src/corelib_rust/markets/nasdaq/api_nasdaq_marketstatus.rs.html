12/// Fetches the current Nasdaq market status information.
13///
14/// This function performs an asynchronous request to the official Nasdaq API
15/// endpoint (`/api/market-info`). It automatically injects the necessary spoofed
16/// headers and validates the application-level `rCode` via the resilient
17/// `nasdaq_end_point` utility.
18///
19/// # Returns
20/// A `Result` containing the parsed `serde_json::Value` of the status `data` payload
21/// on success, or an error message string on failure.
22pub async fn get_status() -> Result<Value, String> {
23    // Forward the request to the extended version using the default Nasdaq base URL
24    get_status_ext("https://api.nasdaq.com").await
25}
26
27/// Internal implementation of status fetching that allows overriding the base URL for testing.
28///
29/// # Arguments
30/// * `base_url` - The base URL for the API request.
31///
32/// # Returns
33/// A `Result` containing the parsed `serde_json::Value` of the `data` payload or an error string.
34async fn get_status_ext(base_url: &str) -> Result<Value, String> {
35    // Construct the full Nasdaq API endpoint URL
36    let url = format!("{}/api/market-info", base_url);
37
38    // Execute the request using the highly resilient `api_nasdaq_unlimited` module
39    let response = nasdaq_end_point::<Value>(&url, None).await;
40
41    // Evaluate the response and map the discriminated union to a standard Result
42    match response {
43        ApiResponse::Success { value } => {
44            // Extract the "data" field from the response body if it exists, otherwise return the full body
45            let data = value.body.get("data").cloned().unwrap_or(value.body);
46            Ok(data)
47        }
48        // Map the structured error reason to a string for the caller
49        ApiResponse::Error { reason } => Err(reason.to_string()),
50    }
51}

64    #[tokio::test]
65    async fn test_get_status_success() {
66        // Start a mock server to simulate the Nasdaq API
67        let server = MockServer::start().await;
68
69        // Define the expected JSON payload
70        let expected_data = json!({
71            "marketIndicator": "Market Open",
72            "mrktStatus": "Open",
73            "isBusinessDay": true,
74            "nextTradeDate": "Mar 10, 2026"
75        });
76
77        // Configure the mock endpoint to return a valid 200 response with rCode 200
78        Mock::given(method("GET"))
79            .and(path("/api/market-info"))
80            .respond_with(ResponseTemplate::new(200).set_body_json(json!({
81                "data": expected_data,
82                "status": { "rCode": 200, "bCodeMessage": null, "developerMessage": null }
83            })))
84            .mount(&server)
85            .await;
86
87        // Execute the internal function against the mock server
88        let res = get_status_ext(&server.uri()).await;
89
90        // Verify the response is successful
91        assert!(res.is_ok(), "Failed to fetch status: {:?}", res);
92
93        // Verify the payload matches expectations
94        let data = res.unwrap();
95        assert_eq!(data["mrktStatus"], "Open");
96        assert_eq!(data["isBusinessDay"], true);
97    }
