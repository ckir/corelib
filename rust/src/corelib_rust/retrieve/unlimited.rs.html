17/// Standardized structure for serialized HTTP responses.
18/// 
19/// Mirrors the `SerializedResponse<T>` interface from TypeScript.
20#[derive(Debug, Clone, Serialize, Deserialize, PartialEq)]
21pub struct SerializedResponse<T> {
22    /// True if the HTTP status code is 2xx
23    pub ok: bool,
24    /// HTTP status code
25    pub status: u16,
26    /// HTTP status text/reason
27    pub status_text: String,
28    /// HTTP headers
29    pub headers: HashMap<String, String>,
30    /// The final URL after any redirects
31    pub url: String,
32    /// The parsed response body
33    pub body: T,
34}
35
36/// Discriminated union for API results, mapping exactly to `{"status": "success", "value": ...}` 
37/// or `{"status": "error", "reason": ...}` for Serde serialization.
38#[derive(Debug, Clone, Serialize, Deserialize, PartialEq)]
39#[serde(tag = "status", rename_all = "lowercase")]
40pub enum ApiResponse<T> {
41    /// Represents a successful HTTP request with a 2xx status code and successfully parsed body.
42    Success { 
43        /// The serialized response wrapper containing the parsed body
44        value: SerializedResponse<T> 
45    },
46    /// Represents a failure, either at the network level, HTTP level (non-2xx), or parsing level.
47    Error { 
48        /// Detailed reason for the error, structured as a JSON value
49        reason: Value 
50    },
51}

53/// Request options to configure individual requests.
54/// 
55/// Defines overrides for method, headers, payload, and ky resilience features.
56#[derive(Debug, Clone, Default)]
57pub struct RequestOptions {
58    /// HTTP Method (Defaults to GET)
59    pub method: Option<Method>,
60    /// Custom HTTP headers
61    pub headers: Option<HashMap<String, String>>,
62    /// JSON payload for POST/PUT/PATCH requests
63    pub json: Option<Value>,
64    /// Request timeout
65    pub timeout: Option<Duration>,
66    /// Maximum number of retry attempts
67    pub retry_limit: Option<u32>,
68}

70/// Makes an HTTP request to a single URL with resilience features.
71///
72/// Uses the underlying `ky` implementation to automatically handle retries and timeouts.
73/// 
74/// # Arguments
75/// * `url` - The target URL to fetch.
76/// * `options` - Optional configuration overrides (method, headers, body, retries).
77/// 
78/// # Returns
79/// An `ApiResponse` containing either the `SerializedResponse` or an error reason.
80pub async fn end_point<T: DeserializeOwned>(
81    url: &str,
82    options: Option<RequestOptions>,
83) -> ApiResponse<T> {
84    let opts = options.unwrap_or_default();
85    let method = opts.method.unwrap_or(Method::GET);
86
87    // Initialize the appropriate ky builder based on the method
88    let mut builder = match method {
89        Method::POST => ky::post(url),
90        Method::PUT => ky::put(url),
91        Method::PATCH => ky::patch(url),
92        Method::DELETE => ky::delete(url),
93        _ => ky::get(url),
94    };
95
96    // Apply headers
97    if let Some(headers) = opts.headers {
98        for (k, v) in headers {
99            builder = builder.header(k, v);
100        }
101    }
102
103    // Apply JSON body
104    if let Some(json_body) = opts.json {
105        builder = builder.json(&json_body);
106    }
107
108    // Apply timeout override
109    if let Some(t) = opts.timeout {
110        builder = builder.timeout(t);
111    }
112
113    // Apply retry limit override
114    if let Some(r) = opts.retry_limit {
115        builder = builder.retry(r);
116    }
117
118    // Execute the request via ky
119    let result = builder.send().await;
120
121    match result {
122        Ok(ky_res) => {
123            // Extract properties before consuming the response body
124            let inner_res = ky_res.into_inner();
125            let status = inner_res.status();
126            let ok = status.is_success();
127            let status_code = status.as_u16();
128            let status_text = status.canonical_reason().unwrap_or("").to_string();
129            let res_url = inner_res.url().to_string();
130            
131            let mut headers = HashMap::new();
132            for (k, v) in inner_res.headers() {
133                headers.insert(
134                    k.as_str().to_string(),
135                    v.to_str().unwrap_or("").to_string(),
136                );
137            }
138
139            // Attempt to parse the body as T
140            match inner_res.json::<T>().await {
141                Ok(body) => ApiResponse::Success {
142                    value: SerializedResponse {
143                        ok,
144                        status: status_code,
145                        status_text,
146                        headers,
147                        url: res_url,
148                        body,
149                    },
150                },
151                Err(e) => ApiResponse::Error {
152                    reason: serde_json::json!({
153                        "message": "Failed to parse response body",
154                        "error": e.to_string(),
155                        "status": status_code,
156                    }),
157                },
158            }
159        }
160        Err(e) => {
161            // Handle request or HTTP level errors
162            let reason = match e {
163                ky::KyError::Http { status, url } => {
164                    serde_json::json!({
165                        "message": format!("HTTP Error {}", status),
166                        "status": status.as_u16(),
167                        "url": url,
168                    })
169                }
170                ky::KyError::Request(req_err) => {
171                    serde_json::json!({
172                        "message": "Transport Error",
173                        "error": req_err.to_string(),
174                    })
175                }
176                ky::KyError::Serialization(ser_err) => {
177                    serde_json::json!({
178                        "message": "Serialization Error",
179                        "error": ser_err.to_string(),
180                    })
181                }
182            };
183            
184            ApiResponse::Error { reason }
185        }
186    }
187}

189/// Makes parallel HTTP requests to multiple URLs.
190/// 
191/// Executes all requests simultaneously using `futures::future::join_all`.
192/// 
193/// # Arguments
194/// * `urls` - A slice of target URLs.
195/// * `options` - Shared configuration overrides applied to all requests.
196/// 
197/// # Returns
198/// A vector of `ApiResponse` objects corresponding to the input order.
199pub async fn end_points<T: DeserializeOwned>(
200    urls: &[&str],
201    options: Option<RequestOptions>,
202) -> Vec<ApiResponse<T>> {
203    let opts = options.unwrap_or_default();
204    
205    // Map each URL to a future calling end_point
206    let futures = urls.iter().map(|&url| {
207        // Clone options for each future
208        let cloned_opts = opts.clone();
209        async move {
210            end_point::<T>(url, Some(cloned_opts)).await
211        }
212    });
213
214    // Execute all futures concurrently
215    join_all(futures).await
216}
